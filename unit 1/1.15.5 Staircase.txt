public class Staircase extends SuperKarel
{    
    public void run()
    {
        putBall();
        while(frontIsClear())
        {
            turnLeft();
            while (ballsPresent()) 
            {
                move();
            }
            
            turnRight();
            move();
            createStep();
        }
    }
    
    private void createStep()
    {
        turnRight();
        putBall();
        while (frontIsClear())
        {
            move();
            putBall();
        }
        turnLeft();
    }
}
